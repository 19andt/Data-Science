def get_parent_range(row):
    if 'Himalaya' in row['Range']:
        return 'Himalaya'
    elif 'Karakoram' in row['Range']:
        return 'Karakoram'
    else:
        return np.NaN

data['Parent Range']=data.apply(get_parent_range, axis=1)
