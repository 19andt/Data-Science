betting_odds=['B365']

def get_prediction(row):
    odds=[row['B365H'], row['B365D'], row['B365A']]
    probs=[1/o for o in odds]
    norm=sum(probs)
    probs=[p/norm for p in probs]
    
    return entropy(probs)
